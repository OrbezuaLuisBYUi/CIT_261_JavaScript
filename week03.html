JSON Parse

function informationPerson(){
var name = prompt("Enter your name");
var lastName = prompt("Enter your last name");
var country = prompt( "Enter your country");
var age = prompt( "Enter age");

//Json format var myInfo = '{"name":"'+name+'", "lastName":"'+lastName+'","country":"'+country+'", "age":"'+age+'"}';
var obj = JSON.parse(myInfo) ;
document.getElementById("resultJson").innerHTML = " My name is: " + obj.name + " " + obj.lastName + " My country " + "is: " + obj.country + " I am " + obj.age + " yeas old ";

var json = {
"0": {
"name": name,
"lastName": lastName,
"country": country,
"age":age
},

"1":{
"name": "Luis",
"lastName": "Peterland",
"country": "USA",
"age":1000
}
};
var persons = "";
for (var i in json) { // i is the index for the Json
for (var j in json[i]) // j is the values , columns
{
persons += " Position: " +i+ " KEY: "+j+" VALUE: "+json[i][j];
}
}
document.getElementById("resultJson2").innerHTML = persons;
